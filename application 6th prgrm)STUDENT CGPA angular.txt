STUDENT CGPA
AIM:
To develop a angular JS application detail of student and they read and number of students and  display the account.
ALGORITHM:
STEP 1:
Create a new project name in command prompt
STEP 2:
Generate the components in ng serve
STEP 3:
Create the component in our project file
STEP 4:
Running the program in ng serve
CODING:
Line 1 import this code app.component.html 
<!DOCTYPE html>
<html ng-app="studentApp">
<head>
  <meta charset="UTF-8">
  <title>Student CGPA App</title>
  <!-- AngularJS CDN -->
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>
</head>
<body ng-controller="StudentController">

  <h1>Student Details & CGPA</h1>

  <!-- Input: Number of Students -->
  <label>Enter number of students: </label>
  <input type="number" ng-model="studentCount" min="1" />
  <button ng-click="generateForm()">Generate Form</button>

  <div ng-if="students.length > 0">
    <h3>Enter Student Details:</h3>
    <div ng-repeat="student in students track by $index">
      <p>
        <strong>Student {{$index + 1}}:</strong><br>
        Name: <input type="text" ng-model="student.name" placeholder="Enter Name" />
        Register No: <input type="text" ng-model="student.regno" placeholder="Enter Reg No" />
        CGPA: <input type="number" step="0.01" ng-model="student.cgpa" placeholder="Enter CGPA" />
      </p>
    </div>
    <button ng-click="showDetails()">Show Details</button>
  </div>

  <!-- Output -->
  <div ng-if="display">
    <h2>Student List (Total: {{students.length}})</h2>
    <table border="1" cellpadding="5">
      <tr>
        <th>S.No</th>
        <th>Name</th>
        <th>Register No</th>
        <th>CGPA</th>
      </tr>
      <tr ng-repeat="s in students track by $index">
        <td>{{$index + 1}}</td>
        <td>{{s.name}}</td>
        <td>{{s.regno}}</td>
        <td>{{s.cgpa}}</td>
      </tr>
    </table>
  </div>

  <script>
    // AngularJS App
    var app = angular.module("studentApp", []);

    app.controller("StudentController", function($scope) {
      $scope.students = [];
      $scope.display = false;

      // Generate empty student forms
      $scope.generateForm = function() {
        $scope.students = [];
        for (var i = 0; i < $scope.studentCount; i++) {
          $scope.students.push({ name: "", regno: "", cgpa: "" });
        }
        $scope.display = false;
      };

      // Show details
      $scope.showDetails = function() {
        $scope.display = true;
      };
    });
  </script>

</body>
</html>
