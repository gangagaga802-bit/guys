<body>
  <input id="userName" value="Annie">
  <input id="userEmail" value="Annie@mail.com">
  <button onclick="updateUser()">Save</button>
  <p id="displayName">Annie</p>
  <p id="displayEmail">Annie@mail.com</p>
  <p id="changeLog"></p>

  <script>
    let oldName = "Annie", oldEmail = "Annie@mail.com";
    function updateUser() {
      let newName = userName.value;
      let newEmail = userEmail.value;
      fetch("data:application/json," + JSON.stringify({ name: newName, email: newEmail }))
        .then(response => response.json())
        .then(updated => {
          displayName.textContent = updated.name;
          displayEmail.textContent = updated.email;
          changeLog.textContent = `Name: ${oldName} ? ${updated.name} | Email: ${oldEmail} ? ${updated.email}`;
          oldName = updated.name;
          oldEmail = updated.email;
        });
    }
  </script>
