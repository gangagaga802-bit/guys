Import React, { useState } from ‘react’;

Const subjects = [‘Tamil’, ‘Math’, ‘Science’, ‘History’, ‘English’];

Function App() {
  Const [hasVoted, setHasVoted] = useState(false);
  Const [message, setMessage] = useState(‘’);
  Const [messageType, setMessageType] = useState(‘’); // ‘success’ or ‘error’
  Const [selectedSubject, setSelectedSubject] = useState(‘’);

  Const handleVote = (subject) => {
    If (hasVoted) {
      setMessage(`? You have already voted for “${selectedSubject}”.`);
      setMessageType(‘error’);
    } else {
      setSelectedSubject(subject);
      setHasVoted(true);
      setMessage(`? Successfully voted for “${subject}”.`);
      setMessageType(‘success’);
    }
  };

  Return (
    <div className=”container”>
      <h1>??? Vote for a Subject</h1>
      <ul className=”subject-list”>
        {subjects.map((subject, index) => (
          <li key={index} className=”subject-item”>
            <button onClick={() => handleVote(subject)} className=”vote-button”>
              Vote for {subject}
            </button>
          </li>
        ))}
      </ul>

      {message && (
        <div className={messageType === ‘success’ ? ‘message-success’ : ‘message-error’}>
          {message}
        </div>
      )}
    </div>
  );
}

Export default App;
