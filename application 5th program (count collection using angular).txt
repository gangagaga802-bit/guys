Line 3 import this coding collection.component.specs.ts 
collection: { name: string }[] = [ 
{ name: 'Milk' }, 
{ name: 'Bread' }, 
{ name: 'Eggs' } 
]; 
// Add new item addItem(newItem: string) { if (newItem.trim()) { 
this.collection.push({ name: newItem }); 
} 
} 
// Remove item by index removeItem(index: number) { this.collection.splice(index, 1); 
} 
// Item count getter 
get itemCount(): number { return this.collection.length; 
} 
}
