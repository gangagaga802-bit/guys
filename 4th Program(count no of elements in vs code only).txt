  <script>
    function countFormElements(form) {
      const all = Array.from(form.elements);
      const total = all.length;

      const controls = all.filter(el => {
        const tag = el.tagName.toLowerCase();
        if (tag === 'fieldset') return false;
        if (el.type === 'submit' || el.type === 'reset' || el.type === 'button') return false;
        return true;
      });

      const inputs = controls.filter(e => e.tagName.toLowerCase() === 'input').length;
      const selects = controls.filter(e => e.tagName.toLowerCase() === 'select').length;
      const textareas = controls.filter(e => e.tagName.toLowerCase() === 'textarea').length;

      const typeCounts = controls.reduce((acc, el) => {
        if (el.tagName.toLowerCase() === 'input') {
          const t = el.type || 'text';
          acc[t] = (acc[t] || 0) + 1;
        }
        return acc;
      }, {});

      return { total, controlsCount: controls.length, inputs, selects, textareas, typeCounts };
    }

    document.addEventListener('DOMContentLoaded', () => {
      const form = document.getElementById('regForm');
      const result = document.getElementById('result');
      const countBtn = document.getElementById('countBtn');

      function render() {
        const stats = countFormElements(form);
        result.innerHTML = `
          <p><strong>Total entries in form.elements:</strong> ${stats.total}</p>
          <p><strong>Form controls (excluding submit/reset/button/fieldset):</strong> ${stats.controlsCount}</p>
          <p><strong>By tag:</strong> inputs=${stats.inputs}, selects=${stats.selects}, textareas=${stats.textareas}</p>
          <p><strong>Input type counts:</strong> ${JSON.stringify(stats.typeCounts)}</p>
        `;
      }
      countBtn.addEventListener('click', render);
      form.addEventListener('submit', e => {
        e.preventDefault();
        alert('Submitting is prevented in this demo.');
      });
    });
  </script>
