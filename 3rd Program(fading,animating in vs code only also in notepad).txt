script.js

$(document).ready(function () {
  $('#fade-in').click(function () {
    $('.gallery-img').fadeIn(1000); // Show image with fade
  });

  $('#fade-out').click(function () {
    $('.gallery-img').fadeOut(1000); // Hide image with fade
  });

  $('#animate').click(function () {
    $('.img-text').animate({
      left: '200px'
    }, 1000).animate({
      left: '10px'
    }, 1000); // Move right, then back
  });
});
