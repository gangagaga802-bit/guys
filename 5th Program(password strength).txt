Write a program to create and build a password, Strength Check using J query
<!DOCTYPE html>
<html>
<head>
<title>Simple Checker</title>
<style> p {
font-size: 12px;
}
</style>
</head>
<body>
<center>
<h3>Password Strength Checker</h3>
<input type="password" id="pass" placeholder="Type password" />
<p>"@" and "#" are the special characters allowed</p>
<p id="msg"></p>
</center>
<script>
const input = document.getElementById("pass");
const msgElement = document.getElementById("msg"); input.onkeyup = function () {
const p = input.value; let msg = "";
let color = "";
if (p.length >= 8 && (p.includes("@") || p.includes("#"))) {

msg = "Good"; color = "green";
} else if (p.length >= 8) { msg = "Okay";
color = "orange";
} else {
msg = "Weak"; color = "red";
}
msgElement.innerText = "Strength: " + msg; msgElement.style.color = color;
};
</script>
</body>
</html>
