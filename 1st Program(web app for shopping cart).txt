  <script>
    const cartItems = document.getElementById('cart-items');
    let itemsInCart = [];

    function addToCart(name, price) {
      if (itemsInCart.length === 0) {
        cartItems.innerHTML = '';
      }

      const existingItem = itemsInCart.find(item => item.name === name);
      if (existingItem) {
        existingItem.qty++;
      } else {
        itemsInCart.push({ name, price, qty: 1 });
      }

      renderCart();
    }

    function removeFromCart(name) {
      itemsInCart = itemsInCart.filter(item => item.name !== name);
      if (itemsInCart.length === 0) {
        cartItems.innerHTML = '<li>No items added.</li>';
      } else {
        renderCart();
      }
    }

    function renderCart() {
      cartItems.innerHTML = '';
      itemsInCart.forEach(item => {
        const li = document.createElement('li');
        li.textContent = `${item.name} - $${item.price} x ${item.qty}`;

        const removeBtn = document.createElement('button');
        removeBtn.textContent = 'Remove';
        removeBtn.className = 'remove-btn';
        removeBtn.onclick = () => removeFromCart(item.name);

        li.appendChild(removeBtn);
        cartItems.appendChild(li);
      });
    }
  </script>
